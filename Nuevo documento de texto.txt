function BuscarCliente(Nacionalidad: Char;Cedula : Longint) : Longint;
Var
Existe: boolean;

incio
  Existe:= false;
  {Archivo Cliente Lidotel }
  {Asigno el Archivo a Leer}
  Assign(ClienteLidotel_Int,ClienteLidotel_Ext);
  {Abrir el Archivo }
  Reset(ClienteLidotel_Int);
  while (eof(ClienteLidotel_Int)=False) And (Existe=False)
    do inicio
      leer(ClienteLidotel_Int,RegClienteLidotel);
      si (RegClienteLidotel.NacionCliente=Nacionalidad) y (RegClienteLidotel.CedulaCliente=Cedula) entonces Existe:= True;
  fin;
  si Existe=true
     entonces PosPuntero:= filePos(ClienteLidotel_Int) -1  {posicion -1, porque los registros arrancan desde cero en el archivo, y la funcion filepos desde 1}
     si no  PosPuntero:= -1;                             {No existe registro forzo -1, para saber que no hay coincidencia}

BuscarCliente:=PosPuntero;
{cierro archivo}
Cerrar(ClienteLidotel_Int);
fin;


Procedure ConsultarCliente;
  var
  dato: string;
    inicio
      dato:='';
        TituloLidotel;
        escribir(' CONSULTAR CLIENTE:');
        escribir;
        escribir;
  {Validar Nacionalidad}
    repetir
      TituloLidotel;
      escribir(' CONSULTAR CLIENTE:');
      escribir;
      escribir(' Nacionalidad (V,E): ');
      NacionCliente:=Upcase(Readkey);
      hasta(NacionCliente='V') o (NacionCliente='E');
      escribirn();
    {Validar Cedula}
      repetir 
      TituloLidotel;
      escribir(' CONSULTAR CLIENTE:');
      escribir;
      escribir(' Nacionalidad (V,E): ', NacionCliente);
      escribir(' Cedula: ');
      leer(dato);
      val(dato,CedulaCliente);
      hasta (CedulaCliente<>0);

     {Validar que la cedula exista para consultar}
      PosCursor:=BuscarCliente(NacionCliente,CedulaCliente);
    si (PosCursor) >= 0 entonces  {Existe Registro }
     begin 
      TituloLidotel;
      escribir(' CONSULTAR CLIENTE:');
      CargarDatosCliente;
      VerDatosCliente;
      escribir();
      escribir();
      escribir(' Pulse Una Tecla Para Salir');
      readkey();
     fin 
     si no 
     inicio 
       escribir;escribir;
       escribir('                      Cliente no Existe...Pulse una tecla para continuar' );
       readkey();
     fin;
   fin;

        ClienteLidotel_Ext_Tmp:='ClienteLidotelTmp.Dat';

        {Verifico si el archivo no existe para crearlo}
        {Crear el Archivo}
        escribir(ClienteLidotel_Int_Tmp);
        {Lectura del Archivo Actual, para enviar registros al Temporal, Exectuando el Cliente Eliminado}
         {Archivo Cliente Lidotel }
         {Asigno el Archivo a Leer}
         Assign(ClienteLidotel_Int,ClienteLidotel_Ext);
         {Abrir el Archivo }
         Reset(ClienteLidotel_Int);
          (eof(ClienteLidotel_Int)=False)
           do inicio
             leer(ClienteLidotel_Int,RegClienteLidotel);
             si  (RegClienteLidotel.NacionCliente<>NacionCliente) y (RegClienteLidotel.CedulaCliente<>CedulaCliente) entonces

        {Cierro Archivo Temporal}
        cerrar(ClienteLidotel_Int_Tmp);
        {Cierro Archivo Original}
        Cerrar(ClienteLidotel_Int);
        {Borro Archivo Original de Clientes}
        erase(ClienteLidotel_Int);
